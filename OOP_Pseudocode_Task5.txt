CLASS Circle INHERITS Shape
    ATTRIBUTES
        PRIVATE radius : NUMBER

    CONSTRUCTOR(c : TEXT, x : NUMBER, y : NUMBER, r : NUMBER)
        CALL Shape.CONSTRUCTOR(c, x, y)
        SET this.radius TO r
    END CONSTRUCTOR

    METHODS
        GET_AREA()
            RETURN 3.14 * radius * radius
        END METHOD

        GET_PERIMETER()
            RETURN 2 * 3.14 * radius
        END METHOD
END CLASS

CLASS Rectangle INHERITS Shape
    ATTRIBUTES
        PRIVATE width : NUMBER
        PRIVATE height : NUMBER

    CONSTRUCTOR(c : TEXT, x : NUMBER, y : NUMBER, w : NUMBER, h : NUMBER)
        CALL Shape.CONSTRUCTOR(c, x, y)
        SET this.width TO w
        SET this.height TO h
    END CONSTRUCTOR

    METHODS
        GET_AREA()
            RETURN width * height
        END METHOD

        GET_PERIMETER()
            RETURN 2 * (width + height)
        END METHOD
END CLASS

CLASS Triangle INHERITS Shape
    ATTRIBUTES
        PRIVATE sideA : NUMBER
        PRIVATE sideB : NUMBER
        PRIVATE sideC : NUMBER

    CONSTRUCTOR(c : TEXT, x : NUMBER, y : NUMBER, a : NUMBER, b : NUMBER, side : NUMBER)
        CALL Shape.CONSTRUCTOR(c, x, y)
        SET this.sideA TO a
        SET this.sideB TO b
        SET this.sideC TO side
    END CONSTRUCTOR

    METHODS
        GET_PERIMETER()
            RETURN sideA + sideB + sideC
        END METHOD

        GET_AREA()
            SET semiPerimeter TO GET_PERIMETER() / 2

            RETURN SQUARE_ROOT(
                semiPerimeter *
                (semiPerimeter - sideA) *
                (semiPerimeter - sideB) *
                (semiPerimeter - sideC)
            )
        END METHOD
END CLASS

CLASS Canvas
    ATTRIBUTES
        PRIVATE canvasName : TEXT
        PRIVATE shapes : LIST<Shape>

    CONSTRUCTOR(name : TEXT)
        SET this.canvasName TO name
        SET this.shapes TO LIST []
    END CONSTRUCTOR

    METHODS
        ADD_SHAPE(s : Shape)
            ADD s TO shapes
            DISPLAY "Shape added to canvas"
        END METHOD

        GET_SHAPE_COUNT()
            RETURN SIZE OF shapes
        END METHOD

        GET_TOTAL_AREA()
            SET totalArea TO 0

            REPEAT FROM I = 1 TO SIZE OF shapes
                SET S TO shapes[I]
                SET totalArea TO totalArea + S.GET_AREA()
            END REPEAT

            RETURN totalArea
        END METHOD

        GET_TOTAL_PERIMETER()
            SET totalPerimeter TO 0

            REPEAT FROM I = 1 TO SIZE OF shapes
                SET S TO shapes[I]
                SET totalPerimeter TO totalPerimeter + S.GET_PERIMETER()
            END REPEAT

            RETURN totalPerimeter
        END METHOD

        DISPLAY_SHAPES()
            REPEAT FROM I = 1 TO SIZE OF shapes
                SET S TO shapes[I]

                DISPLAY "Color: " + S.GET_COLOR()
                DISPLAY "Position: " + S.GET_POSITION_X() + ", " + S.GET_POSITION_Y()
                DISPLAY "Area: " + S.GET_AREA()
                DISPLAY "Perimeter: " + S.GET_PERIMETER()
            END REPEAT
        END METHOD
END CLASS

CLASS PaymentMethod

        PRIVATE amount : NUMBER
        PRIVATE status : TEXT

    CONSTRUCTOR(a : NUMBER)
        SET this.amount TO a
        SET this.status TO "Pending"
    END CONSTRUCTOR

    METHODS
        GET_AMOUNT()
            RETURN amount
        END METHOD

        GET_STATUS()
            RETURN status
        END METHOD

        PROCESS_PAYMENT()
            DISPLAY "Payment processing method not defined"
        END METHOD
END CLASS

CLASS CreditCardPayment INHERITS PaymentMethod
    ATTRIBUTES
        PRIVATE cardNumber : TEXT
        PRIVATE expiryDate : TEXT

    CONSTRUCTOR(a : NUMBER, card : TEXT, expiry : TEXT)
        CALL PaymentMethod.CONSTRUCTOR(a)
        SET this.cardNumber TO card
        SET this.expiryDate TO expiry
    END CONSTRUCTOR

    METHODS
        IS_CARD_VALID()
            IF LENGTH OF cardNumber EQUALS 16 AND expiryDate NOT EQUALS "" THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        PROCESS_PAYMENT()
            IF IS_CARD_VALID() THEN
                SET status TO "Completed"
                DISPLAY "Credit card payment completed"
            ELSE
                DISPLAY "Payment refused: invalid card details"
            END IF
        END METHOD
END CLASS

CLASS BankTransferPayment INHERITS PaymentMethod
    ATTRIBUTES
        PRIVATE accountNumber : TEXT
        PRIVATE delayCompleted : BOOLEAN

    CONSTRUCTOR(a : NUMBER, account : TEXT)
        CALL PaymentMethod.CONSTRUCTOR(a)
        SET this.accountNumber TO account
        SET this.delayCompleted TO FALSE
    END CONSTRUCTOR

    METHODS
        IS_ACCOUNT_VALID()
            IF accountNumber NOT EQUALS "" THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        COMPLETE_DELAY()
            SET delayCompleted TO TRUE
        END METHOD

        PROCESS_PAYMENT()
            IF NOT IS_ACCOUNT_VALID() THEN
                DISPLAY "Payment refused: invalid account number"
            ELSE
                IF NOT delayCompleted THEN
                    DISPLAY "Bank transfer pending: waiting for transfer delay"
                ELSE
                    SET status TO "Completed"
                    DISPLAY "Bank transfer payment completed"
                END IF
            END IF
        END METHOD
END CLASS

CLASS WalletPayment INHERITS PaymentMethod
    ATTRIBUTES
        PRIVATE walletName : TEXT
        PRIVATE walletBalance : NUMBER

    CONSTRUCTOR(a : NUMBER, name : TEXT, balance : NUMBER)
        CALL PaymentMethod.CONSTRUCTOR(a)
        SET this.walletName TO name
        SET this.walletBalance TO balance
    END CONSTRUCTOR

    METHODS
        GET_BALANCE()
            RETURN walletBalance
        END METHOD

        PROCESS_PAYMENT()
            IF walletBalance LESS THAN amount THEN
                DISPLAY "Payment refused: insufficient wallet balance"
            ELSE
                SET walletBalance TO walletBalance - amount
                SET status TO "Completed"
                DISPLAY "Wallet payment completed"
            END IF
        END METHOD
END CLASS

CLASS Order
    ATTRIBUTES
        orderNumber : TEXT
        items : LIST<OrderItem>
        payment : PaymentMethod

    CONSTRUCTOR(number : TEXT)
        SET orderNumber TO number
        SET items TO LIST []
        SET payment TO NULL
    END CONSTRUCTOR

    METHODS
        ADD_ITEM(item : OrderItem)
            ADD item TO items
        END METHOD

        GET_TOTAL()
            SET total TO 0

            REPEAT FROM I = 1 TO SIZE OF items
                SET I_ITEM TO items[I]
                SET total TO total + I_ITEM.GET_TOTAL_PRICE()
            END REPEAT

            RETURN total
        END METHOD

        HAS_PAYMENT()
            IF payment EQUALS NULL THEN
                RETURN FALSE
            ELSE
                RETURN TRUE
            END IF
        END METHOD

        CHECKOUT_WITH_CREDIT_CARD(card : TEXT, expiry : TEXT)
            IF HAS_PAYMENT() THEN
                DISPLAY "Checkout refused: order already has a payment method"
            ELSE
                CREATE OBJECT newPayment OF CreditCardPayment(
                    GET_TOTAL(),
                    card,
                    expiry
                )

                SET payment TO newPayment
                payment.PROCESS_PAYMENT()
            END IF
        END METHOD

        CHECKOUT_WITH_BANK_TRANSFER(account : TEXT)
            IF HAS_PAYMENT() THEN
                DISPLAY "Checkout refused: order already has a payment method"
            ELSE
                CREATE OBJECT newPayment OF BankTransferPayment(
                    GET_TOTAL(),
                    account
                )

                SET payment TO newPayment
                payment.PROCESS_PAYMENT()
            END IF
        END METHOD

        CHECKOUT_WITH_WALLET(walletName : TEXT, balance : NUMBER)
            IF HAS_PAYMENT() THEN
                DISPLAY "Checkout refused: order already has a payment method"
            ELSE
                CREATE OBJECT newPayment OF WalletPayment(
                    GET_TOTAL(),
                    walletName,
                    balance
                )

                SET payment TO newPayment
                payment.PROCESS_PAYMENT()
            END IF
        END METHOD

        GET_PAYMENT()
            RETURN payment
        END METHOD
END CLASS

CLASS UserAccount
    ATTRIBUTES
        userName : TEXT
        emailAddress : TEXT
        subscription : Subscription

    CONSTRUCTOR(name : TEXT, email : TEXT, sub : Subscription)
        SET userName TO name
        SET emailAddress TO email
        SET subscription TO sub
        SET watchHistory TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN userName
        END METHOD

        GET_SUBSCRIPTION()
            RETURN subscription
        END METHOD

        CHANGE_SUBSCRIPTION(newSubscription : Subscription)
            SET subscription TO newSubscription
            DISPLAY "Subscription changed successfully"
        END METHOD

        ADD_WATCH_RECORD(title : TEXT, date : TEXT, currentDate : TEXT)
            IF NOT subscription.IS_ACTIVE(currentDate) THEN
                DISPLAY "Watch refused: subscription is not active"
            ELSE
                CREATE OBJECT newRecord OF WatchRecord(title, date)
                ADD newRecord TO watchHistory

                DISPLAY title + " added to watch history"
            END IF
        END METHOD

        GET_WATCH_COUNT()
            RETURN SIZE OF watchHistory
        END METHOD

            DISPLAY "Monthly Price: " + subscription.GET_MONTHLY_PRICE()
            DISPLAY "Streaming Quality: " + subscription.GET_STREAMING_QUALITY()
            DISPLAY "Device Limit: " + subscription.GET_DEVICE_LIMIT()
            DISPLAY "Downloads Allowed: " + subscription.ALLOWS_DOWNLOADS()
            DISPLAY "Active: " + subscription.IS_ACTIVE(currentDate)
        END METHOD

        DISPLAY_WATCH_HISTORY()
            REPEAT FROM I = 1 TO SIZE OF watchHistory
                SET W TO watchHistory[I]

                DISPLAY W.GET_TITLE() + " watched on " + W.GET_WATCHED_DATE()
            END REPEAT
        END METHOD

CLASS DebitCard INHERITS Card
    ATTRIBUTES
        linkedAccount : BankAccount
        purchases : LIST<Purchase>

    CONSTRUCTOR(number : TEXT, expiry : TEXT, securityData : TEXT, account : BankAccount)
        CALL Card.CONSTRUCTOR(number, expiry, securityData)
        SET linkedAccount TO account
        SET purchases TO LIST []
    END CONSTRUCTOR

    METHODS
        AUTHORIZE_PURCHASE(description : TEXT, amount : NUMBER, date : TEXT, securityData : TEXT)
            IF NOT IS_ACTIVE() THEN
                DISPLAY "Purchase refused: card is blocked"
            ELSE
                IF NOT chip.VALIDATE_SECURITY_DATA(securityData) THEN
                    DISPLAY "Purchase refused: invalid chip security data"
                ELSE
                    IF amount GREATER THAN linkedAccount.GET_AVAILABLE_BALANCE() THEN
                        DISPLAY "Purchase refused: insufficient account balance"
                    ELSE
                        linkedAccount.WITHDRAW(amount)

                        CREATE OBJECT newPurchase OF Purchase(
                            description,
                            amount,
                            date
                        )

                        newPurchase.MARK_AS_PAID()
                        ADD newPurchase TO purchases

                        DISPLAY "Debit card purchase authorized"
                    END IF
                END IF
            END IF
        END METHOD
END CLASS

CLASS CreditCard INHERITS Card
    ATTRIBUTES
        creditLimit : NUMBER
        purchases : LIST<Purchase>

    CONSTRUCTOR(number : TEXT, expiry : TEXT, securityData : TEXT, limit : NUMBER)
        CALL Card.CONSTRUCTOR(number, expiry, securityData)
        SET creditLimit TO limit
        SET purchases TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_OUTSTANDING_BALANCE()
            SET total TO 0

            REPEAT FROM I = 1 TO SIZE OF purchases
                SET P TO purchases[I]

                IF NOT P.IS_PAID() THEN
                    SET total TO total + P.GET_AMOUNT()
                END IF
            END REPEAT

            RETURN total
        END METHOD

        GET_AVAILABLE_CREDIT()
            RETURN creditLimit - GET_OUTSTANDING_BALANCE()
        END METHOD

        AUTHORIZE_PURCHASE(description : TEXT, amount : NUMBER, date : TEXT, securityData : TEXT)
            IF NOT IS_ACTIVE() THEN
                DISPLAY "Purchase refused: card is blocked"
            ELSE
                IF NOT chip.VALIDATE_SECURITY_DATA(securityData) THEN
                    DISPLAY "Purchase refused: invalid chip security data"
                ELSE
                    IF amount GREATER THAN GET_AVAILABLE_CREDIT() THEN
                        DISPLAY "Purchase refused: credit limit exceeded"
                    ELSE
                        CREATE OBJECT newPurchase OF Purchase(
                            description,
                            amount,
                            date
                        )

                        ADD newPurchase TO purchases

                        DISPLAY "Credit card purchase authorized"
                    END IF
                END IF
            END IF
        END METHOD

        PAY_PURCHASE(p : Purchase)
            p.MARK_AS_PAID()
            DISPLAY "Purchase payment recorded"
        END METHOD
END CLASS

    CONSTRUCTOR(t : TEXT, a : TEXT, pages : NUMBER, cover : CoverImage, totalTime : NUMBER)

    END CONSTRUCTOR

    METHODS

            SET this.listenedTime TO this.listenedTime + minutes
        END METHOD

        GET_PROGRESS()
            RETURN (this.listenedTime * 100) / this.totalListeningTime
        END METHOD
END CLASS

CLASS Device
    ATTRIBUTES
        PRIVATE deviceName : TEXT
        PRIVATE roomLocation : TEXT
        PRIVATE isOn : BOOLEAN

    CONSTRUCTOR(name : TEXT, room : TEXT)
        SET this.deviceName TO name
        SET this.roomLocation TO room
        SET this.isOn TO FALSE
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN this.deviceName
        END METHOD

        GET_ROOM_LOCATION()
            RETURN this.roomLocation
        END METHOD

        IS_ON()
            RETURN this.isOn
        END METHOD

        TURN_ON()
            SET this.isOn TO TRUE
            DISPLAY this.deviceName + " turned on"
        END METHOD

        TURN_OFF()
            SET this.isOn TO FALSE
            DISPLAY this.deviceName + " turned off"
        END METHOD

        GET_POWER_USAGE()
            RETURN 0
        END METHOD
END CLASS

CLASS SmartLight INHERITS Device
    ATTRIBUTES
        PRIVATE brightnessLevel : NUMBER
        PRIVATE maximumPower : NUMBER

    CONSTRUCTOR(name : TEXT, room : TEXT, maxPower : NUMBER)
        CALL Device.CONSTRUCTOR(name, room)
        SET this.brightnessLevel TO 0
        SET this.maximumPower TO maxPower
    END CONSTRUCTOR

    METHODS
        SET_BRIGHTNESS(level : NUMBER)
            IF level LESS THAN 0 OR level GREATER THAN 100 THEN
                DISPLAY "Brightness refused: level must be between 0 and 100"
            ELSE
                SET this.brightnessLevel TO level
            END IF
        END METHOD

        GET_BRIGHTNESS()
            RETURN this.brightnessLevel
        END METHOD

        GET_POWER_USAGE()
            IF NOT IS_ON() THEN
                RETURN 0
            ELSE
                RETURN this.maximumPower * (this.brightnessLevel / 100)
            END IF
        END METHOD
END CLASS

CLASS SmartThermostat INHERITS Device
    ATTRIBUTES
        PRIVATE currentTemperature : NUMBER
        PRIVATE targetTemperature : NUMBER
        PRIVATE maximumPower : NUMBER

    CONSTRUCTOR(name : TEXT, room : TEXT, currentTemp : NUMBER, targetTemp : NUMBER, maxPower : NUMBER)
        CALL Device.CONSTRUCTOR(name, room)
        SET this.currentTemperature TO currentTemp
        SET this.targetTemperature TO targetTemp
        SET this.maximumPower TO maxPower
    END CONSTRUCTOR

    METHODS
        SET_TARGET_TEMPERATURE(temp : NUMBER)
            SET this.targetTemperature TO temp
        END METHOD

        SET_CURRENT_TEMPERATURE(temp : NUMBER)
            SET this.currentTemperature TO temp
        END METHOD

        GET_TARGET_TEMPERATURE()
            RETURN this.targetTemperature
        END METHOD

        GET_POWER_USAGE()
            IF NOT IS_ON() THEN
                RETURN 0
            ELSE
                SET temperatureDifference TO ABSOLUTE_VALUE(
                    this.targetTemperature - this.currentTemperature
                )

                IF temperatureDifference GREATER THAN OR EQUAL TO 10 THEN
                    RETURN this.maximumPower
                ELSE
                    RETURN this.maximumPower * (temperatureDifference / 10)
                END IF
            END IF
        END METHOD
END CLASS

CLASS StorageUnit
    ATTRIBUTES
        PRIVATE totalCapacity : NUMBER
        PRIVATE usedSpace : NUMBER

    CONSTRUCTOR(capacity : NUMBER)
        IF capacity GREATER THAN 0 THEN
            SET this.totalCapacity TO capacity
        ELSE
            SET this.totalCapacity TO 1
            DISPLAY "Invalid capacity: default set to 1"
        END IF

        SET this.usedSpace TO 0
    END CONSTRUCTOR

    METHODS
        GET_TOTAL_CAPACITY()
            RETURN this.totalCapacity
        END METHOD

        GET_USED_SPACE()
            RETURN this.usedSpace
        END METHOD

        GET_REMAINING_SPACE()
            RETURN this.totalCapacity - this.usedSpace
        END METHOD

        IS_FULL()
            IF this.usedSpace GREATER THAN OR EQUAL TO this.totalCapacity THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        SAVE_FOOTAGE(spaceRequired : NUMBER)
            IF spaceRequired LESS THAN OR EQUAL TO 0 THEN
                DISPLAY "Recording refused: invalid footage size"
                RETURN FALSE
            ELSE
                IF spaceRequired GREATER THAN GET_REMAINING_SPACE() THEN
                    DISPLAY "Recording refused: storage is full"
                    RETURN FALSE
                ELSE
                    SET this.usedSpace TO this.usedSpace + spaceRequired
                    DISPLAY "Footage saved successfully"
                    RETURN TRUE
                END IF
            END IF
        END METHOD
END CLASS

CLASS SmartCamera INHERITS Device
    ATTRIBUTES
        PRIVATE isRecording : BOOLEAN
        PRIVATE recordingPower : NUMBER
        PRIVATE standbyPower : NUMBER
        PRIVATE storage : StorageUnit

    CONSTRUCTOR(
        name : TEXT,
        room : TEXT,
        recordingUsage : NUMBER,
        standbyUsage : NUMBER,
        storageCapacity : NUMBER
    )
        CALL Device.CONSTRUCTOR(name, room)
        SET this.isRecording TO FALSE
        SET this.recordingPower TO recordingUsage
        SET this.standbyPower TO standbyUsage
        CREATE OBJECT this.storage OF StorageUnit(storageCapacity)
    END CONSTRUCTOR

    METHODS
        START_RECORDING(footageSize : NUMBER)
            IF NOT IS_ON() THEN
                DISPLAY "Recording refused: camera is off"
            ELSE
                IF this.storage.IS_FULL() THEN
                    DISPLAY "Recording refused: storage is full"
                ELSE
                    IF this.storage.SAVE_FOOTAGE(footageSize) THEN
                        SET this.isRecording TO TRUE
                        DISPLAY this.deviceName + " started recording"
                    END IF
                END IF
            END IF
        END METHOD

        STOP_RECORDING()
            SET this.isRecording TO FALSE
            DISPLAY "Camera stopped recording"
        END METHOD

        IS_RECORDING()
            RETURN this.isRecording
        END METHOD

        GET_REMAINING_STORAGE()
            RETURN this.storage.GET_REMAINING_SPACE()
        END METHOD

        GET_POWER_USAGE()
            IF NOT IS_ON() THEN
                RETURN 0
            ELSE
                IF this.isRecording THEN
                    RETURN this.recordingPower
                ELSE
                    RETURN this.standbyPower
                END IF
            END IF
        END METHOD
END CLASS

CLASS Room
    ATTRIBUTES
        PRIVATE roomName : TEXT
        PRIVATE devices : LIST<Device>

    CONSTRUCTOR(name : TEXT)
        SET this.roomName TO name
        SET this.devices TO LIST []
    END CONSTRUCTOR

    METHODS
        ADD_DEVICE(d : Device)
            ADD d TO this.devices
            DISPLAY d.GET_NAME() + " added to " + this.roomName
        END METHOD

        GET_DEVICE_COUNT()
            RETURN SIZE OF this.devices
        END METHOD

        GET_TOTAL_POWER_USAGE()
            SET totalUsage TO 0

            REPEAT FROM I = 1 TO SIZE OF this.devices
                SET D TO this.devices[I]
                SET totalUsage TO totalUsage + D.GET_POWER_USAGE()
            END REPEAT

            RETURN totalUsage
        END METHOD

        DISPLAY_DEVICES()
            REPEAT FROM I = 1 TO SIZE OF this.devices
                SET D TO this.devices[I]

                DISPLAY "Device: " + D.GET_NAME()
                DISPLAY "Location: " + D.GET_ROOM_LOCATION()
                DISPLAY "Power usage: " + D.GET_POWER_USAGE()
            END REPEAT
        END METHOD
END CLASS
