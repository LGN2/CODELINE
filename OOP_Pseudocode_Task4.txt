CLASS Attempt
    ATTRIBUTES
        exam : Exam
        student : Student
        startTime : NUMBER
        selectedAnswers : LIST<SelectedAnswer>

    CONSTRUCTOR(e : Exam, s : Student, start : NUMBER)
        SET exam TO e
        SET student TO s
        SET startTime TO start
        SET selectedAnswers TO LIST []
    END CONSTRUCTOR

    METHODS
        HAS_TIME_EXPIRED(currentTime : NUMBER)
            IF currentTime - startTime GREATER THAN OR EQUAL TO exam.GET_TIME_LIMIT() THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        HAS_ANSWERED(q : Question)
            REPEAT FROM I = 1 TO SIZE OF selectedAnswers
                SET A TO selectedAnswers[I]

                IF A.GET_QUESTION() EQUALS q THEN
                    RETURN TRUE
                END IF
            END REPEAT

            RETURN FALSE
        END METHOD

        SUBMIT_ANSWER(q : Question, answer : TEXT, currentTime : NUMBER)
            IF HAS_TIME_EXPIRED(currentTime) THEN
                DISPLAY "Answer refused: exam time limit has passed"
            ELSE
                IF HAS_ANSWERED(q) THEN
                    DISPLAY "Answer refused: question already answered"
                ELSE
                    CREATE OBJECT selectedAnswer OF SelectedAnswer(q, answer)
                    ADD selectedAnswer TO selectedAnswers
                    DISPLAY "Answer submitted successfully"
                END IF
            END IF
        END METHOD

        GET_SCORE()
            SET score TO 0

            REPEAT FROM I = 1 TO SIZE OF selectedAnswers
                SET A TO selectedAnswers[I]
                SET Q TO A.GET_QUESTION()

                IF Q.IS_CORRECT(A.GET_ANSWER()) THEN
                    SET score TO score + Q.GET_POINT_VALUE()
                END IF
            END REPEAT

            RETURN score
        END METHOD

        GET_EXAM()
            RETURN exam
        END METHOD
END CLASS

CLASS MeetingRoom
    ATTRIBUTES
        roomName : TEXT
        capacity : NUMBER
        bookings : LIST<Booking>

    CONSTRUCTOR(name : TEXT, cap : NUMBER)
        SET roomName TO name
        SET capacity TO cap
        SET bookings TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN roomName
        END METHOD

        IS_AVAILABLE(date : TEXT, start : NUMBER, end : NUMBER)
            REPEAT FROM I = 1 TO SIZE OF bookings
                SET B TO bookings[I]

                IF B.GET_DATE() EQUALS date THEN
                    IF start LESS THAN B.GET_END_TIME() AND end GREATER THAN B.GET_START_TIME() THEN
                        RETURN FALSE
                    END IF
                END IF
            END REPEAT

            RETURN TRUE
        END METHOD

        ADD_BOOKING(b : Booking)
            IF IS_AVAILABLE(
                b.GET_DATE(),
                b.GET_START_TIME(),
                b.GET_END_TIME()
            ) THEN
                ADD b TO bookings
                DISPLAY "Room booking added successfully"
            ELSE
                DISPLAY "Booking refused: room already booked during this time"
            END IF
        END METHOD
END CLASS

CLASS Member
    ATTRIBUTES
        memberName : TEXT
        phoneNumber : TEXT
        membershipPlan : MembershipPlan
        bookings : LIST<Booking>

    CONSTRUCTOR(name : TEXT, phone : TEXT, plan : MembershipPlan)
        SET memberName TO name
        SET phoneNumber TO phone
        SET membershipPlan TO plan
        SET bookings TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN memberName
        END METHOD

        GET_USED_HOURS(month : TEXT)
            SET totalHours TO 0

            REPEAT FROM I = 1 TO SIZE OF bookings
                SET B TO bookings[I]

                IF B.GET_DATE() CONTAINS month THEN
                    SET totalHours TO totalHours + B.GET_DURATION()
                END IF
            END REPEAT

            RETURN totalHours
        END METHOD

        GET_REMAINING_HOURS(month : TEXT)
            RETURN membershipPlan.GET_MONTHLY_QUOTA() - GET_USED_HOURS(month)
        END METHOD

        BOOK_ROOM(room : MeetingRoom, date : TEXT, start : NUMBER, end : NUMBER, month : TEXT)
            SET requestedHours TO end - start

            IF requestedHours LESS THAN OR EQUAL TO 0 THEN
                DISPLAY "Booking refused: invalid time slot"
            ELSE
                IF requestedHours GREATER THAN GET_REMAINING_HOURS(month) THEN
                    DISPLAY "Booking refused: monthly meeting-room quota exceeded"
                ELSE
                    IF NOT room.IS_AVAILABLE(date, start, end) THEN
                        DISPLAY "Booking refused: room already booked during this time"
                    ELSE
                        CREATE OBJECT newBooking OF Booking(
                            THIS,
                            room,
                            date,
                            start,
                            end
                        )

                        ADD newBooking TO bookings
                        room.ADD_BOOKING(newBooking)

                        DISPLAY "Booking created successfully for " + memberName
                    END IF
                END IF
            END IF
        END METHOD
END CLASS
