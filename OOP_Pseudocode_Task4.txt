        SET this.customers TO LIST []

                DISPLAY "Genre: " + B.GET_GENRE_NAME()

CLASS Question

        PRIVATE questionText : TEXT
        PRIVATE correctAnswer : TEXT
        PRIVATE pointValue : NUMBER

    CONSTRUCTOR(text : TEXT, answer : TEXT, points : NUMBER)
        SET this.questionText TO text
        SET this.correctAnswer TO answer

        IF points GREATER THAN 0 THEN
            SET this.pointValue TO points
        ELSE
            SET this.pointValue TO 0
            DISPLAY "Invalid point value"
        END IF
    END CONSTRUCTOR

    METHODS
        GET_TEXT()
            RETURN this.questionText
        END METHOD

        GET_POINT_VALUE()
            RETURN this.pointValue
        END METHOD

        IS_CORRECT(answer : TEXT)
            IF answer EQUALS this.correctAnswer THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD
END CLASS

CLASS SelectedAnswer
    ATTRIBUTES
        PRIVATE question : Question
        PRIVATE answerText : TEXT

    CONSTRUCTOR(q : Question, answer : TEXT)
        SET this.question TO q
        SET this.answerText TO answer
    END CONSTRUCTOR

    METHODS
        GET_QUESTION()
            RETURN this.question
        END METHOD

        GET_ANSWER()
            RETURN this.answerText
        END METHOD
END CLASS

CLASS Exam
    ATTRIBUTES
        PRIVATE examName : TEXT
        PRIVATE timeLimitMinutes : NUMBER
        PRIVATE questions : LIST<Question>

    CONSTRUCTOR(name : TEXT, limitMinutes : NUMBER)
        SET this.examName TO name
        SET this.timeLimitMinutes TO limitMinutes
        SET this.questions TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN this.examName
        END METHOD

        GET_TIME_LIMIT_MINUTES()
            RETURN this.timeLimitMinutes
        END METHOD

        ADD_QUESTION(q : Question)
            ADD q TO this.questions
        END METHOD

        CONTAINS_QUESTION(q : Question)
            REPEAT FROM I = 1 TO SIZE OF this.questions
                IF this.questions[I] EQUALS q THEN
                    RETURN TRUE
                END IF
            END REPEAT

            RETURN FALSE
        END METHOD

        GET_TOTAL_POINTS()
            SET total TO 0

            REPEAT FROM I = 1 TO SIZE OF this.questions
                SET Q TO this.questions[I]
                SET total TO total + Q.GET_POINT_VALUE()
            END REPEAT

            RETURN total
        END METHOD
END CLASS

CLASS Attempt
    ATTRIBUTES
        PRIVATE exam : Exam
        PRIVATE student : Student
        PRIVATE startTimeMinutes : NUMBER
        PRIVATE selectedAnswers : LIST<SelectedAnswer>

    CONSTRUCTOR(e : Exam, s : Student, startMinutes : NUMBER)
        SET this.exam TO e
        SET this.student TO s
        SET this.startTimeMinutes TO startMinutes
        SET this.selectedAnswers TO LIST []
    END CONSTRUCTOR

    METHODS
        HAS_TIME_EXPIRED(currentTimeMinutes : NUMBER)
            SET elapsedMinutes TO currentTimeMinutes - this.startTimeMinutes

            IF elapsedMinutes GREATER THAN OR EQUAL TO this.exam.GET_TIME_LIMIT_MINUTES() THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        HAS_ANSWERED(q : Question)
            REPEAT FROM I = 1 TO SIZE OF this.selectedAnswers
                SET A TO this.selectedAnswers[I]

                IF A.GET_QUESTION() EQUALS q THEN
                    RETURN TRUE
                END IF
            END REPEAT

            RETURN FALSE
        END METHOD

        SUBMIT_ANSWER(q : Question, answer : TEXT, currentTimeMinutes : NUMBER)
            IF NOT this.exam.CONTAINS_QUESTION(q) THEN
                DISPLAY "Answer refused: question does not belong to this exam"
            ELSE
                IF HAS_TIME_EXPIRED(currentTimeMinutes) THEN
                    DISPLAY "Answer refused: exam time limit has passed"
                ELSE
                    IF HAS_ANSWERED(q) THEN
                        DISPLAY "Answer refused: question already answered"
                    ELSE
                        CREATE OBJECT selectedAnswer OF SelectedAnswer(q, answer)
                        ADD selectedAnswer TO this.selectedAnswers
                        DISPLAY "Answer submitted successfully"
                    END IF
                END IF
            END IF
        END METHOD

        GET_SCORE()
            SET score TO 0

            REPEAT FROM I = 1 TO SIZE OF this.selectedAnswers
                SET A TO this.selectedAnswers[I]
                SET Q TO A.GET_QUESTION()

                IF Q.IS_CORRECT(A.GET_ANSWER()) THEN
                    SET score TO score + Q.GET_POINT_VALUE()
                END IF
            END REPEAT

            RETURN score
        END METHOD
END CLASS

        SET this.drivers TO LIST []

                IF D.GET_NAME() EQUALS name THEN

                END IF

                        driver.SET_AVAILABLE(FALSE)
                        ADD newTrip TO this.trips

                        DISPLAY "Trip created successfully"
                    END IF
                END IF
            END IF

CLASS MeetingRoom
    ATTRIBUTES
        PRIVATE roomName : TEXT
        PRIVATE capacity : NUMBER
        PRIVATE bookings : LIST<Booking>

    CONSTRUCTOR(name : TEXT, cap : NUMBER)
        SET this.roomName TO name

        IF cap GREATER THAN 0 THEN
            SET this.capacity TO cap
        ELSE
            SET this.capacity TO 1
            DISPLAY "Invalid capacity: default set to 1"
        END IF

        SET this.bookings TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN this.roomName
        END METHOD

        GET_CAPACITY()
            RETURN this.capacity
        END METHOD

        IS_AVAILABLE(date : TEXT, start : NUMBER, end : NUMBER)
            REPEAT FROM I = 1 TO SIZE OF this.bookings
                SET B TO this.bookings[I]

                IF B.GET_DATE() EQUALS date THEN
                    IF start LESS THAN B.GET_END_TIME() AND end GREATER THAN B.GET_START_TIME() THEN
                        RETURN FALSE
                    END IF
                END IF
            END REPEAT

            RETURN TRUE
        END METHOD

        ADD_BOOKING(b : Booking)
            ADD b TO this.bookings
        END METHOD

        DISPLAY_BOOKINGS()
            REPEAT FROM I = 1 TO SIZE OF this.bookings
                SET B TO this.bookings[I]

                DISPLAY B.GET_DATE()
                DISPLAY B.GET_START_TIME() + " to " + B.GET_END_TIME()
            END REPEAT
        END METHOD
END CLASS

CLASS Member
    ATTRIBUTES
        PRIVATE memberName : TEXT
        PRIVATE phoneNumber : TEXT
        PRIVATE membershipPlan : MembershipPlan
        PRIVATE bookings : LIST<Booking>

    CONSTRUCTOR(name : TEXT, phone : TEXT, plan : MembershipPlan)
        SET this.memberName TO name
        SET this.phoneNumber TO phone
        SET this.membershipPlan TO plan
        SET this.bookings TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN this.memberName
        END METHOD

        EXTRACT_MONTH(date : TEXT)
            RETURN FIRST 7 CHARACTERS OF date
        END METHOD

        GET_USED_HOURS(month : TEXT)
            SET totalHours TO 0

            REPEAT FROM I = 1 TO SIZE OF this.bookings
                SET B TO this.bookings[I]

                IF EXTRACT_MONTH(B.GET_DATE()) EQUALS month THEN
                    SET totalHours TO totalHours + B.GET_DURATION_HOURS()
                END IF
            END REPEAT

            RETURN totalHours
        END METHOD

        GET_REMAINING_HOURS(month : TEXT)
            RETURN this.membershipPlan.GET_MONTHLY_QUOTA() - GET_USED_HOURS(month)
        END METHOD

        BOOK_ROOM(room : MeetingRoom, date : TEXT, start : NUMBER, end : NUMBER, peopleCount : NUMBER)
            SET requestedHours TO end - start
            SET month TO EXTRACT_MONTH(date)

            IF requestedHours LESS THAN OR EQUAL TO 0 THEN
                DISPLAY "Booking refused: invalid time slot"
            ELSE
                IF peopleCount GREATER THAN room.GET_CAPACITY() THEN
                    DISPLAY "Booking refused: room capacity exceeded"
                ELSE
                    IF requestedHours GREATER THAN GET_REMAINING_HOURS(month) THEN
                        DISPLAY "Booking refused: monthly quota exceeded"
                    ELSE
                        IF NOT room.IS_AVAILABLE(date, start, end) THEN
                            DISPLAY "Booking refused: room already booked"
                        ELSE
                            CREATE OBJECT newBooking OF Booking(
                                THIS,
                                room,
                                date,
                                start,
                                end
                            )

                            ADD newBooking TO this.bookings
                            room.ADD_BOOKING(newBooking)

                            DISPLAY "Booking created successfully"
                        END IF
                    END IF
                END IF
            END IF
        END METHOD

        DISPLAY_BOOKINGS()
            REPEAT FROM I = 1 TO SIZE OF this.bookings
                SET B TO this.bookings[I]

                DISPLAY "Room: " + B.GET_ROOM_NAME()
                DISPLAY "Date: " + B.GET_DATE()
                DISPLAY "Duration: " + B.GET_DURATION_HOURS() + " hours"
            END REPEAT
        END METHOD
END CLASS

CLASS Client
    ATTRIBUTES
        PRIVATE clientName : TEXT
        PRIVATE phoneNumber : TEXT
        PRIVATE emailAddress : TEXT
        PRIVATE sessions : LIST<FitnessSession>

    CONSTRUCTOR(name : TEXT, phone : TEXT, email : TEXT)
        SET this.clientName TO name
        SET this.phoneNumber TO phone
        SET this.emailAddress TO email
        SET this.sessions TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN this.clientName
        END METHOD

        ADD_SESSION(s : FitnessSession)
            ADD s TO this.sessions
        END METHOD

        REMOVE_SESSION(s : FitnessSession)
            REMOVE s FROM this.sessions
        END METHOD

        GET_SESSION_COUNT()
            RETURN SIZE OF this.sessions
        END METHOD
END CLASS

CLASS FitnessSession
    ATTRIBUTES
        PRIVATE client : Client
        PRIVATE sessionDate : TEXT
        PRIVATE startTime : NUMBER
        PRIVATE durationMinutes : NUMBER
        PRIVATE workoutType : TEXT

    CONSTRUCTOR(c : Client, date : TEXT, time : NUMBER, duration : NUMBER, type : TEXT)
        SET this.client TO c
        SET this.sessionDate TO date
        SET this.startTime TO time
        SET this.durationMinutes TO duration
        SET this.workoutType TO type
    END CONSTRUCTOR

    METHODS
        GET_CLIENT()
            RETURN this.client
        END METHOD

        GET_DATE()
            RETURN this.sessionDate
        END METHOD

        GET_START_TIME()
            RETURN this.startTime
        END METHOD

        GET_END_TIME()
            RETURN this.startTime + this.durationMinutes
        END METHOD

        GET_DURATION()
            RETURN this.durationMinutes
        END METHOD

        GET_WORKOUT_TYPE()
            RETURN this.workoutType
        END METHOD

        RESCHEDULE(newDate : TEXT, newTime : NUMBER)
            SET this.sessionDate TO newDate
            SET this.startTime TO newTime
        END METHOD
END CLASS

CLASS PersonalTrainer
    ATTRIBUTES
        PRIVATE trainerName : TEXT
        PRIVATE specialties : LIST<TEXT>
        PRIVATE bookedSessions : LIST<FitnessSession>

    CONSTRUCTOR(name : TEXT)
        SET this.trainerName TO name
        SET this.specialties TO LIST []
        SET this.bookedSessions TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN this.trainerName
        END METHOD

        ADD_SPECIALTY(type : TEXT)
            ADD type TO this.specialties
        END METHOD

        HAS_SPECIALTY(type : TEXT)
            REPEAT FROM I = 1 TO SIZE OF this.specialties
                IF this.specialties[I] EQUALS type THEN
                    RETURN TRUE
                END IF
            END REPEAT

            RETURN FALSE
        END METHOD

        IS_AVAILABLE(date : TEXT, start : NUMBER, duration : NUMBER)
            SET requestedEnd TO start + duration

            REPEAT FROM I = 1 TO SIZE OF this.bookedSessions
                SET S TO this.bookedSessions[I]

                IF S.GET_DATE() EQUALS date THEN
                    IF start LESS THAN S.GET_END_TIME() AND requestedEnd GREATER THAN S.GET_START_TIME() THEN
                        RETURN FALSE
                    END IF
                END IF
            END REPEAT

            RETURN TRUE
        END METHOD

        BOOK_SESSION(client : Client, date : TEXT, time : NUMBER, duration : NUMBER, type : TEXT)
            IF date EQUALS "" THEN
                DISPLAY "Booking refused: date cannot be empty"
            ELSE
                IF duration LESS THAN OR EQUAL TO 0 THEN
                    DISPLAY "Booking refused: duration must be greater than zero"
                ELSE
                    IF NOT HAS_SPECIALTY(type) THEN
                        DISPLAY "Booking refused: workout is outside trainer specialty"
                    ELSE
                        IF NOT IS_AVAILABLE(date, time, duration) THEN
                            DISPLAY "Booking refused: trainer is already booked"
                        ELSE
                            CREATE OBJECT newSession OF FitnessSession(
                                client,
                                date,
                                time,
                                duration,
                                type
                            )

                            ADD newSession TO this.bookedSessions
                            client.ADD_SESSION(newSession)

                            DISPLAY "Session booked successfully"
                        END IF
                    END IF
                END IF
            END IF
        END METHOD

        CANCEL_SESSION(session : FitnessSession)
            REMOVE session FROM this.bookedSessions
            session.GET_CLIENT().REMOVE_SESSION(session)
            DISPLAY "Session cancelled successfully"
        END METHOD

        RESCHEDULE_SESSION(session : FitnessSession, newDate : TEXT, newTime : NUMBER)
            IF IS_AVAILABLE(
                newDate,
                newTime,
                session.GET_DURATION()
            ) THEN
                session.RESCHEDULE(newDate, newTime)
                DISPLAY "Session rescheduled successfully"
            ELSE
                DISPLAY "Reschedule refused: trainer is unavailable"
            END IF
        END METHOD

        DISPLAY_SCHEDULE_BY_DATE(date : TEXT)
            REPEAT FROM I = 1 TO SIZE OF this.bookedSessions
                SET S TO this.bookedSessions[I]

                IF S.GET_DATE() EQUALS date THEN
                    DISPLAY "Client: " + S.GET_CLIENT().GET_NAME()
                    DISPLAY "Time: " + S.GET_START_TIME()
                    DISPLAY "Workout: " + S.GET_WORKOUT_TYPE()
                END IF
            END REPEAT
        END METHOD

        GET_TOTAL_SESSION_COUNT()
            RETURN SIZE OF this.bookedSessions
        END METHOD
END CLASS
