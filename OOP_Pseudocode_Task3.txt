CLASS Examination
    ATTRIBUTES
        diagnosis : TEXT
        vitalSigns : LIST<VitalSign>
        treatments : LIST<Treatment>

    CONSTRUCTOR(d : TEXT)
        SET diagnosis TO d
        SET vitalSigns TO LIST []
        SET treatments TO LIST []
    END CONSTRUCTOR

    METHODS
        ADD_VITAL_SIGN(v : VitalSign)
            ADD v TO vitalSigns
        END METHOD

        ADD_TREATMENT(t : Treatment)
            ADD t TO treatments
        END METHOD

        IS_URGENT()
            REPEAT FROM I = 1 TO SIZE OF vitalSigns
                SET V TO vitalSigns[I]

                IF NOT V.IS_SAFE() THEN
                    RETURN TRUE
                END IF
            END REPEAT

            RETURN FALSE
        END METHOD

        GET_TREATMENT_TOTAL()
            SET total TO 0

            REPEAT FROM I = 1 TO SIZE OF treatments
                SET T TO treatments[I]
                SET total TO total + T.GET_COST()
            END REPEAT

            RETURN total
        END METHOD
END CLASS

CLASS Screen
    ATTRIBUTES
        screenNumber : NUMBER
        capacity : NUMBER
        seatingLayout : TEXT
        showtimes : LIST<Showtime>

    CONSTRUCTOR(num : NUMBER, cap : NUMBER, layout : TEXT)
        SET screenNumber TO num
        SET capacity TO cap
        SET seatingLayout TO layout
        SET showtimes TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_CAPACITY()
            RETURN capacity
        END METHOD

        HAS_OVERLAP(start : TEXT, end : TEXT)
            REPEAT FROM I = 0 TO SIZE OF showtimes -1
                SET S TO showtimes[I]

                IF start LESS THAN S.GET_END_TIME() AND end GREATER THAN S.GET_START_TIME() THEN
                    RETURN TRUE
                END IF
            END REPEAT

            RETURN FALSE
        END METHOD

        ADD_SHOWTIME(s : Showtime)
            IF HAS_OVERLAP(s.GET_START_TIME(), s.GET_END_TIME()) THEN
                DISPLAY "Showtime refused: overlaps another showtime on the same screen"
            ELSE
                ADD s TO showtimes
                DISPLAY "Showtime added successfully"
            END IF
        END METHOD
END CLASS

CLASS Showtime
    ATTRIBUTES
        film : Film
        screen : Screen
        startTime : TEXT
        endTime : TEXT
        bookedTickets : LIST<Ticket>

    CONSTRUCTOR(f : Film, sc : Screen, start : TEXT, end : TEXT)
        SET film TO f
        SET screen TO sc
        SET startTime TO start
        SET endTime TO end
        SET bookedTickets TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_START_TIME()
            RETURN startTime
        END METHOD

        GET_END_TIME()
            RETURN endTime
        END METHOD

        GET_REMAINING_SEATS()
            RETURN screen.GET_CAPACITY() - SIZE OF bookedTickets
        END METHOD

        IS_SEAT_BOOKED(seat : TEXT)
            REPEAT FROM I = 0 TO SIZE OF bookedTickets -1
                SET T TO bookedTickets[I]

                IF T.GET_SEAT_NUMBER() EQUALS seat THEN
                    RETURN TRUE
                END IF
            END REPEAT

            RETURN FALSE
        END METHOD

        BOOK_TICKET(customer : Customer, seat : TEXT)
            IF GET_REMAINING_SEATS() LESS THAN OR EQUAL TO 0 THEN
                DISPLAY "Booking refused: screen is full"
            ELSE
                IF IS_SEAT_BOOKED(seat) THEN
                    DISPLAY "Booking refused: seat already booked"
                ELSE
                    CREATE OBJECT newTicket OF Ticket(customer.GET_NAME(), seat, THIS)
                    ADD newTicket TO bookedTickets
                    DISPLAY "Ticket booked successfully for " + customer.GET_NAME()
                END IF
            END IF
        END METHOD
END CLASS

    METHODS
        ADD_PRODUCT(p : Product)
            ADD p TO products
        END METHOD

        RECORD_DELIVERY(isLate : BOOLEAN)
            SET totalDeliveries TO totalDeliveries + 1

            IF isLate THEN
                SET lateDeliveries TO lateDeliveries + 1
            END IF
        END METHOD

            IF totalDeliveries EQUALS 0 THEN

CLASS Product
    ATTRIBUTES
        productName : TEXT
        category : Category
        vendor : Vendor
        bins : LIST<StorageBin>
        movements : LIST<StockMovement>

    CONSTRUCTOR(name : TEXT, cat : Category, v : Vendor)
        SET productName TO name
        SET category TO cat
        SET vendor TO v
        SET bins TO LIST []
        SET movements TO LIST []
    END CONSTRUCTOR

    METHODS
        ADD_BIN(bin : StorageBin)
            ADD bin TO bins
        END METHOD

        GET_TOTAL_STOCK()
            SET total TO 0

            REPEAT FROM I = 1 TO SIZE OF bins
                SET B TO bins[I]
                SET total TO total + B.GET_QUANTITY()
            END REPEAT

            RETURN total
        END METHOD

        MOVE_STOCK_IN(qty : NUMBER, date : TEXT)
            IF SIZE OF bins EQUALS 0 THEN
                DISPLAY "Movement refused: no storage bin available"
            ELSE
                SET B TO bins[1]
                B.ADD_STOCK(qty)

                CREATE OBJECT movement OF StockMovement("IN", qty, date)
                ADD movement TO movements

                DISPLAY "Stock added successfully"
            END IF
        END METHOD

        MOVE_STOCK_OUT(qty : NUMBER, date : TEXT)
            IF qty GREATER THAN GET_TOTAL_STOCK() THEN
                DISPLAY "Movement refused: stock cannot go below zero"
            ELSE
                SET remainingQty TO qty

                REPEAT FROM I = 1 TO SIZE OF bins
                    SET B TO bins[I]

                    IF remainingQty GREATER THAN 0 THEN
                        IF B.GET_QUANTITY() GREATER THAN OR EQUAL TO remainingQty THEN
                            B.REMOVE_STOCK(remainingQty)
                            SET remainingQty TO 0
                        ELSE
                            SET taken TO B.GET_QUANTITY()
                            B.REMOVE_STOCK(taken)
                            SET remainingQty TO remainingQty - taken
                        END IF
                    END IF
                END REPEAT

                CREATE OBJECT movement OF StockMovement("OUT", qty, date)
                ADD movement TO movements

                DISPLAY "Stock removed successfully"
            END IF
        END METHOD

        GET_NAME()
            RETURN productName
        END METHOD
END CLASS

CLASS Student
    ATTRIBUTES
        studentName : TEXT
        studentID : TEXT
        personalRecord : TEXT
        grades : LIST<Grade>
        enrolledClasses : LIST<SchoolClass>

    CONSTRUCTOR(name : TEXT, id : TEXT, record : TEXT)
        SET studentName TO name
        SET studentID TO id
        SET personalRecord TO record
        SET grades TO LIST []
        SET enrolledClasses TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN studentName
        END METHOD

        ADD_GRADE(g : Grade)
            ADD g TO grades
        END METHOD

        GET_OVERALL_AVERAGE()
            SET totalWeightedMarks TO 0
            SET totalWeighting TO 0

            REPEAT FROM I = 1 TO SIZE OF grades
                SET G TO grades[I]
                SET totalWeightedMarks TO totalWeightedMarks + G.GET_WEIGHTED_MARK()
                SET totalWeighting TO totalWeighting + G.GET_WEIGHTING()
            END REPEAT

            IF totalWeighting EQUALS 0 THEN
                RETURN 0
            ELSE
                RETURN totalWeightedMarks / totalWeighting
            END IF
        END METHOD

        HAS_TIME_CONFLICT(time : TEXT)
            REPEAT FROM I = 1 TO SIZE OF enrolledClasses
                SET C TO enrolledClasses[I]

                IF C.GET_SCHEDULED_TIME() EQUALS time THEN
                    RETURN TRUE
                END IF
            END REPEAT

            RETURN FALSE
        END METHOD

        ADD_CLASS(c : SchoolClass)
            ADD c TO enrolledClasses
        END METHOD
END CLASS

CLASS Claim

        claimDetails : TEXT

        claimAmount : NUMBER

    CONSTRUCTOR(details : TEXT, type : TEXT, amount : NUMBER)
        SET claimDetails TO details
        SET coverageType TO type
        SET claimAmount TO amount
        SET status TO "Filed"
    END CONSTRUCTOR

    METHODS
        GET_AMOUNT()
            RETURN claimAmount
        END METHOD

        GET_COVERAGE_TYPE()
            RETURN coverageType
        END METHOD

        GET_STATUS()
            RETURN status
        END METHOD

        MOVE_TO_REVIEW()
            IF status EQUALS "Filed" THEN
                SET status TO "Under Review"
            ELSE
                DISPLAY "Invalid status change"
            END IF
        END METHOD

        APPROVE()
            IF status EQUALS "Under Review" THEN
                SET status TO "Approved"
            ELSE
                DISPLAY "Claim must be under review before approval"
            END IF
        END METHOD

        REJECT()
            IF status EQUALS "Under Review" THEN
                SET status TO "Rejected"
            ELSE
                DISPLAY "Claim must be under review before rejection"
            END IF
        END METHOD

        IS_APPROVED()
            IF status EQUALS "Approved" THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

CLASS Policy
    ATTRIBUTES
        policyNumber : TEXT
        policyholder : Policyholder
        coverages : LIST<Coverage>
        claims : LIST<Claim>

    CONSTRUCTOR(number : TEXT, holder : Policyholder)
        SET policyNumber TO number
        SET policyholder TO holder
        SET coverages TO LIST []
        SET claims TO LIST []
    END CONSTRUCTOR

    METHODS
        ADD_COVERAGE(c : Coverage)
            ADD c TO coverages
        END METHOD

        FIND_COVERAGE_LIMIT(type : TEXT)
            REPEAT FROM I = 1 TO SIZE OF coverages
                SET C TO coverages[I]

                IF C.GET_TYPE() EQUALS type THEN
                    RETURN C.GET_LIMIT()
                END IF
            END REPEAT

            RETURN 0
        END METHOD

        FILE_CLAIM(details : TEXT, type : TEXT, amount : NUMBER)
            SET limit TO FIND_COVERAGE_LIMIT(type)

            IF limit EQUALS 0 THEN
                DISPLAY "Claim refused: coverage type not found"
            ELSE
                IF amount GREATER THAN limit THEN
                    DISPLAY "Claim refused: amount exceeds coverage limit"
                ELSE
                    CREATE OBJECT newClaim OF Claim(details, type, amount)
                    ADD newClaim TO claims
                    DISPLAY "Claim filed successfully"
                END IF
            END IF
        END METHOD

        GET_TOTAL_COVERAGE_LIMIT()
            SET total TO 0

            REPEAT FROM I = 1 TO SIZE OF coverages
                SET C TO coverages[I]
                SET total TO total + C.GET_LIMIT()
            END REPEAT

            RETURN total
        END METHOD

        GET_TOTAL_APPROVED_CLAIMS()
            SET total TO 0

            REPEAT FROM I = 1 TO SIZE OF claims
                SET CL TO claims[I]

                IF CL.IS_APPROVED() THEN
                    SET total TO total + CL.GET_AMOUNT()
                END IF

CLASS Crop
    ATTRIBUTES
        cropName : TEXT
        growingRequirements : TEXT
        waterNeed : NUMBER
        expectedYield : NUMBER

    CONSTRUCTOR(name : TEXT, requirements : TEXT, water : NUMBER, yield : NUMBER)

        SET growingRequirements TO requirements
        SET waterNeed TO water
        SET expectedYield TO yield
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN cropName
        END METHOD

        GET_WATER_NEED()
            RETURN waterNeed
        END METHOD

        GET_EXPECTED_YIELD()
            RETURN expectedYield
        END METHOD
END CLASS

CLASS Activity
    ATTRIBUTES
        activityType : TEXT
        activityDate : TEXT
        amount : NUMBER
        worker : Worker

    CONSTRUCTOR(type : TEXT, date : TEXT, a : NUMBER, w : Worker)
        SET activityType TO type
        SET activityDate TO date
        SET amount TO a
        SET worker TO w
    END CONSTRUCTOR

    METHODS
        GET_TYPE()
            RETURN activityType
        END METHOD

        GET_DATE()
            RETURN activityDate
        END METHOD

        GET_AMOUNT()
            RETURN amount
        END METHOD

        IS_PROPER_CARE(crop : Crop)
            IF activityType EQUALS "Watering" THEN
                IF amount GREATER THAN OR EQUAL TO crop.GET_WATER_NEED() THEN
                    RETURN TRUE
                ELSE
                    RETURN FALSE
                END IF
            ELSE
                IF activityType EQUALS "Fertilizing" AND amount GREATER THAN 0 THEN
                    RETURN TRUE
                ELSE
                    RETURN FALSE
                END IF
            END IF
        END METHOD
END CLASS

CLASS Worker
    ATTRIBUTES
        workerName : TEXT
        workerRecord : TEXT
        dailyActivityLimit : NUMBER
        assignedActivities : LIST<Activity>

    CONSTRUCTOR(name : TEXT, record : TEXT, limit : NUMBER)
        SET workerName TO name
        SET workerRecord TO record
        SET dailyActivityLimit TO limit
        SET assignedActivities TO LIST []
    END CONSTRUCTOR

    METHODS
        GET_NAME()
            RETURN workerName
        END METHOD

        COUNT_ACTIVITIES_ON_DATE(date : TEXT)
            SET count TO 0

            REPEAT FROM I = 1 TO SIZE OF assignedActivities
                SET A TO assignedActivities[I]

                IF A.GET_DATE() EQUALS date THEN
                    SET count TO count + 1
                END IF
            END REPEAT

            RETURN count
        END METHOD

        CAN_ACCEPT_ACTIVITY(date : TEXT)
            IF COUNT_ACTIVITIES_ON_DATE(date) LESS THAN dailyActivityLimit THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        ASSIGN_ACTIVITY(a : Activity)
            IF CAN_ACCEPT_ACTIVITY(a.GET_DATE()) THEN
                ADD a TO assignedActivities
                DISPLAY "Activity assigned to " + workerName
            ELSE
                DISPLAY "Assignment refused: worker daily limit reached"
            END IF
        END METHOD
END CLASS

CLASS Field
    ATTRIBUTES
        fieldName : TEXT
        season : TEXT
        crop : Crop
        activityLog : LIST<Activity>

    CONSTRUCTOR(name : TEXT, s : TEXT, c : Crop)
        SET fieldName TO name
        SET season TO s
        SET crop TO c
        SET activityLog TO LIST []
    END CONSTRUCTOR

    METHODS
        ADD_ACTIVITY(type : TEXT, date : TEXT, amount : NUMBER, worker : Worker)
            IF NOT worker.CAN_ACCEPT_ACTIVITY(date) THEN
                DISPLAY "Activity refused: worker daily limit reached"
            ELSE
                IF type EQUALS "Watering" AND amount LESS THAN crop.GET_WATER_NEED() THEN
                    DISPLAY "Activity refused: water amount is below crop requirement"
                ELSE
                    CREATE OBJECT newActivity OF Activity(type, date, amount, worker)

                    ADD newActivity TO activityLog
                    worker.ASSIGN_ACTIVITY(newActivity)

                    DISPLAY type + " activity logged successfully"
                END IF
            END IF
        END METHOD

        GET_CARE_PERCENTAGE()
            IF SIZE OF activityLog EQUALS 0 THEN
                RETURN 0
            ELSE
                SET properCareCount TO 0

                REPEAT FROM I = 1 TO SIZE OF activityLog
                    SET A TO activityLog[I]

                    IF A.IS_PROPER_CARE(crop) THEN
                        SET properCareCount TO properCareCount + 1
                    END IF
                END REPEAT

                RETURN (properCareCount / SIZE OF activityLog) * 100
            END IF
        END METHOD

        GET_ESTIMATED_YIELD()
            RETURN crop.GET_EXPECTED_YIELD() * (GET_CARE_PERCENTAGE() / 100)
        END METHOD
END CLASS
